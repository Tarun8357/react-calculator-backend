                checkout scm

        }

            steps {
                // Compile and package the application using Maven
                sh 'mvn clean package'
            }
        }
        
        stage('Test') {
            steps {
                // Run tests using Maven
                sh 'mvn test'
            }
        }
        
        stage('Deploy') {
            steps {
                // Here, you can add deployment steps if needed
                // For example, deploying to a server or a container
                // sh 'your deployment command'
            }
        }
    }
    
    post {
        always {
            // Archive the generated artifacts
            archiveArtifacts artifacts: '**/target/*.jar', allowEmptyArchive: true

        success {
            // Send a notification if the pipeline succeeds
            echo 'Pipeline succeeded!'

    }
